-- Creating database
CREATE DATABASE normalizations;
USE normalizations;

-- Creating table Students
CREATE TABLE Students (
    s_id INT PRIMARY KEY,
    name VARCHAR(50),
    section VARCHAR(50)
);

-- Creating table StudentGames
CREATE TABLE StudentGames (
    s_id INT,
    game VARCHAR(50),
    FOREIGN KEY (s_id) REFERENCES Students(s_id)
);

-- Creating table StudentsGames_1NF
CREATE TABLE StudentsGames_1NF (
    s_id INT,
    name VARCHAR(50),
    game VARCHAR(50),
    section VARCHAR(50)
);

-- Creating table StudentsGames_UNF
CREATE TABLE StudentsGames_UNF (
    s_id INT,
    name VARCHAR(50),
    games VARCHAR(100),
    section VARCHAR(50)
);
